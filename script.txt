local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")

local LocalPlayer = Players.LocalPlayer

local TEAM = 1

local GOAL_POSITION = TEAM == 1
    and Vector3.new(-319.546, 6.30647, 91.892)
    or Vector3.new(-110.238, 6.30647, 92.0445)

while true do
    local field = Workspace:WaitForChild("FootballField")
    local ball = field:WaitForChild("SoccerBall")

    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local hrp = character:WaitForChild("HumanoidRootPart")

    -- Wait until the ball touches your character naturally
    repeat
        if not ball.Parent then
            break
        end

        ball.Position = hrp.Position
        task.wait()
    until (ball.Position - hrp.Position).Magnitude < 3

    -- Move the ball to the goal
    while ball.Parent do
        ball.Position = GOAL_POSITION
        task.wait()
    end
end
