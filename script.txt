local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")

local LocalPlayer = Players.LocalPlayer

local TEAM = 1 -- 1 or 2

local GOAL_POSITION = TEAM == 1
	and Vector3.new(-319.546, 6.30647, 91.892)
	or Vector3.new(-110.238, 6.30647, 92.0445)

while true do
	local field = Workspace:WaitForChild("FootballField")
	local ball = field:WaitForChild("SoccerBall")

	local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()

	local connection
	connection = ball.Touched:Connect(function(hit)
		if not hit:IsDescendantOf(character) then
			return
		end

		connection:Disconnect()

		local start = os.clock()
		while os.clock() - start < 1 and ball.Parent do
			ball.Position = GOAL_POSITION
			task.wait(0.01)
		end
	end)

	-- Wait until the current ball is removed before handling the next one
	ball.AncestryChanged:Wait()
end
