local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")

local LocalPlayer = Players.LocalPlayer

local TEAM = 1 -- 1 or 2

local GOAL_POSITION = TEAM == 1
	and Vector3.new(-319.546, 6.30647, 91.892)
	or Vector3.new(-110.238, 6.30647, 92.0445)

while task.wait(0.05) do
	local field = game.Workspace:WaitForChild("FootballField")

	local ball = field:WaitForChild("SoccerBall")
	if not ball then
		continue
	end

	local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
	local hrp = character:FindFirstChild("HumanoidRootPart") or character:WaitForChild("HumanoidRootPart")

	local start = os.clock()
	while ball.Parent and os.clock() - start < 2 do
		ball.CFrame = hrp.CFrame
		task.wait(0.01)
	end
	task.wait(0.05)

	-- move to goal for 3 seconds
	local start = os.clock()
	while ball.Parent and os.clock() - start < 10 do
		ball.CFrame = CFrame.new(GOAL_POSITION) + Vector3.new(math.random(0, 1), math.random(0, 1), math.random(0, 1))
		task.wait(0.01)
	end

	-- wait until this ball is gone before looking for the next one
	while ball.Parent do
		task.wait(0.05)
	end
end
